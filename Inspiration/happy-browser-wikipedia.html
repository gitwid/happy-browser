    let pageLoading = false;

    // ── classifier (rule-based, no generative) ──
    function stripHtml(h){ return h.replace(/<[^>]+>/g,'').replace(/\s+/g,' ').trim(); }
    function wc(html){ const t=stripHtml(html); return t?t.split(/\s+/).length:0; }
    function lc(html){ return (html.match(/<a[\s>]/gi)||[]).length; }
    function tc(html){ return (html.match(/<table[\s>]/gi)||[]).length; }
    function classifySection(sec){
    const title=(sec.line||sec.anchor||'').toLowerCase();
    const text=sec.text||'';
    const words=wc(text), links=lc(text), tables=tc(text);
    if(/^references?$|^notes?$|^citation|^bibliography|^footnote/i.test(title)) return 'footer';
    if(/^see also|^external link|^further reading|^source/i.test(title)) return 'cta';
    if(/^gallery|illustration|^figure/i.test(title)) return 'visual';
    if(tables>0 && words<200) return 'data';
    if(words<20 && links>4) return 'nav';
    return 'semantic';
    }
    function extractLinks(allHtml, limit=6){
    const seen=new Set(), out=[];
    const re=/href="\/wiki\/([^"#:]+)"/g;
    let m;
    while((m=re.exec(allHtml))!==null){
    const slug=m[1];
    const title=decodeURIComponent(slug).replace(/_/g,' ');
    if(!seen.has(slug)&&!/^(File|Help|Wikipedia|Template|Category|Special|Talk|User):/i.test(slug)){
    seen.add(slug); out.push({title,slug,rel:0.9+Math.random()*.1});
    if(out.length>=limit) break;
    }
    }
    return out;
    }
    function buildPageModel(data, slug){
    const secs=[];
    // Always-present Wikipedia nav bar
    secs.push({title:'navigation · wikipedia',type:'nav',wordCount:0,linkCount:12,
    y0:0,y1:0,text:'',wikiSection:false});
    // Lead hero image
    if(data.lead&&data.lead.image){
    secs.push({title:'lead image',type:'visual',wordCount:0,linkCount:0,
    y0:0,y1:0,text:'',wikiSection:false});
    }
    // Lead text
    const leadText=(data.lead&&data.lead.sections&&data.lead.sections[0]&&data.lead.sections[0].text)||'';
    const leadWords=wc(leadText);
    secs.push({title:data.lead&&data.lead.displaytitle||slug.replace(/_/g,' '),
    type:'semantic',wordCount:Math.max(leadWords,40),linkCount:lc(leadText),
    y0:0,y1:0,text:leadText,wikiSection:false,isLead:true});
    // Remaining sections
    const remaining=(data.remaining&&data.remaining.sections)||[];
    let allText=leadText;
    remaining.forEach(function(s){
    const text=s.text||'';
    allText+=text;
    const words=wc(text);
    secs.push({
    title:s.line||s.anchor||'section',
    type:classifySection(s),
    wordCount:Math.max(words,8),
    linkCount:lc(text),
    toclevel:s.toclevel||1,
    y0:0,y1:0,text,wikiSection:true,
    });
    });
    // Compute proportional zones
    const NAV_WEIGHT=0.07, VIS_WEIGHT=0.10, MIN_SEC=0.04;
    let weights=secs.map(function(s){
    if(s.type==='nav') return NAV_WEIGHT;
    if(s.type==='visual') return VIS_WEIGHT;
    return Math.max(MIN_SEC, s.wordCount/600);
    });
    const total=weights.reduce(function(a,b){return a+b;},0);
    let y=0;
    secs.forEach(function(s,i){
    s.y0=y;
    s.y1=Math.min(1,y+weights[i]/total);
    y=s.y1;
    });
    // Clamp last section to 1
    if(secs.length) secs[secs.length-1].y1=1;
    // Internal links for nav rail
    const links=extractLinks(allText);
    // Add current article as "current" slot
    const curTitle=(data.lead&&data.lead.displaytitle)||slug.replace(/_/g,' ');
    links.splice(2,0,{title:curTitle,slug,rel:1.0,current:true});
    return{title:curTitle,sections:secs,links:links.slice(0,6)};
    }
    // ── fetch ──
    async function fetchArticle(slug){
    slug=slug.trim().replace(/\s+/g,'_');
    pageLoading=true; pageError=null; pageModel=null;
    parseY=0; mode='parsing';
    updateModeBtn('browsing');
    setAddrState('fetching…');
    try{
    const url='https://en.wikipedia.org/api/rest_v1/page/mobile-sections/'+encodeURIComponent(slug);
    const res=await fetch(url,{signal:AbortSignal.timeout(6000)});
